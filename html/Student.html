from PlayerInterface import PlayerInterface
import MathQuestion
import savedata

class Student(PlayerInterface):
    """
    A class representing a student user in the game, handling authentication,
    score tracking, and level progression.

    Inherits from PlayerInterface and implements its methods to provide functionality
    specific to student users, such as answering math questions, signing up, signing in,
    and managing score and level progression.

    Attributes:
        username (str): The student's username.
        password (str): The student's password.
        email (str): The student's email address.
        highscore (int or None): The student's high score.

    Methods:
        answer(q: MathQuestion, a: int): Returns True if the answer is correct, False otherwise.
        signup(username: str, password: str, email: str, data: savedata): Registers a new student user.
        signin(username: str, password: str, saves: savedata): Authenticates a student user.
        nextLevel(): Increments the student's high score.
        getScore(): Returns the student's high score.
        getUserName(): Returns the student's username.
    """
    def __init__(self):
        self.username = ""
        self.password = ""
        self.email = ""
        self.highscore = None

    def answer(self, q: MathQuestion, a: int):
        if q.checkAnswer(a):
            return True
        return False

    def signup(self, username: str, password: str, email: str, data: savedata):
        self.username = username
        self.password = password
        self.email = email
        self.highscore = 0
        return data.addNewUser(username, password, email, 'Student')

    def signin(self, username: str, password: str, saves: savedata):
        userData = saves.getUserData(username, password)
        if userData == None:
            return False
        self.username = userData['Username']
        self.password = userData['Password']
        self.email = userData['Email']
        self.highscore = userData['Score']

    def nextLevel(self):
        self.highscore += 1

    def getScore(self):
        return self.highscore

    def getUserName(self):
        return self.username
